// SumIntegers program calculating and printing the sum of two integers

public class SumIntegers { // Define class named SumIntegers
    public static void main(String[] args) { // Main method: entry point of the program
        int num1 = 20; // Declare integer variable num1 and assign 20
        int num2 = 30; // Declare integer variable num2 and assign 30
        int sum = num1 + num2; // Calculate sum of num1 and num2 and store in sum
        System.out.println("Sum of Two Integers: " + sum); // Print the result with label
    } // End of main method
} // End of class SumIntegers
